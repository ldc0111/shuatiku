#### 题解

`````c++
#include <stdio.h>
#include <8.h>


int main(){
  long long max_p, p = 1, zero_num = 0;//累乘变量，０记录，最大值
  for(int i = 8; str[i]; i++){//字符串最后一位等于０，
    //除运算，和去掉０
    if(i >= 13){//累乘１３次
      if (str[i - 13] == '0'){
          zero_num--;
      } else {
        p /= (str[i - 13] - '0');
      }
    }
    //乘运算，和统计０
    if(str[i] == '0'){
      zero_num++;
    } else {
      p *= (str[i] - '0');
    }
    
    if(zero_num == 0 && i >= 12 && p > max_p){//等于０才有效，证明没有０
      max_p = p;
    }
  }
  
  printf("%d\n", max_p);
  return 0;
}
`````
