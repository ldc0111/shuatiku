#### 题目大意

计算一千位的斐波那契数为第几位







#### 运用了两个框架

高精度加法框架

循环数组框架





#### 代码

````c++
#include <stdio.h>

int main(){
  int fib[3][1005] = {0};
  fib[0][0] = fib[0][1] = 1;//初始化位数，和两个值
  fib[1][0] = fib[1][1] = 1;
  int n  = 1;//初始化项数
  while(fib[n % 3][0] < 1000){//上一次计算的项数
    n++;//本次需要计算的项数，
    for(int i = 1; i <= fib[(n - 1) % 3]; i++){//按位加上前两个的每一位，
      fib[n % 3][i] = fib[(n - 1) % 3][i] + fib[(n - 2) % 3][i;
    }
    fib[n % 3][0] = fib[(n - 1) % 3][0];//加上前一个数的位数，因为他的位数最大，
      //高精加法框架                                                       
    for(int i = 1; i <= fib[n % 3][0]; i++){//处理每一位
      if(fib[n % 3][i] < 10) continue;//如果小于１０，不返回
      fib[n % 3][i + 1] += fib[n % 3][i] / 10;//进位数
      fib[n % 3][i] %= 10;//取余数
      if(i == fib[n % 3][0]) fib[n % 3][0]++;//如果最后一位仍需要进位，则位数加１
    }
  }
  
  printf("%d\n", n + 1);//输出
  
  return 0;
}
````
